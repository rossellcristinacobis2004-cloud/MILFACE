224@app.route('/api/verificar_master', methods=['POST'])
225def api_verificar_master():
226    """Recibe un frame, lo compara con las fotos del Máster con LBPH separado."""
227    data = request.json
228    imagen_b64 = data.get('imagen')
229    if not imagen_b64:
230        return {"status": "error"}, 400
231
232    try:
233        header, encoded = imagen_b64.split(',', 1)
234        imagen_data = base64.b64decode(encoded)
235        np_arr = np.frombuffer(imagen_data, np.uint8)
236        frame  = cv2.imdecode(np_arr, cv2.IMREAD_COLOR)
237
238        # Cargar fotos del master
239        conexion = conectar()
240        cursor = conexion.cursor()
241        cursor.execute("SELECT foto_frente, foto_derecha, foto_izquierda FROM master LIMIT 1")
242        fila = cursor.fetchone()
243        conexion.close()
244
245        if not fila:
246            return {"status": "error", "mensaje": "No hay master registrado"}
247
248        import cv2 as _cv2
249        face_cascade_m = _cv2.CascadeClassifier(_cv2.data.haarcascades + 'haarcascade_frontalface_default.xml')
250        recognizer_m = _cv2.face.LBPHFaceRecognizer_create()
251
252        faces_m, labels_m = [], []
253        for idx, ruta in enumerate(fila):
254            if ruta and os.path.exists(ruta):
255                img = _cv2.imread(ruta)
256                if img is not None:
257                    gris = _cv2.cvtColor(img, _cv2.COLOR_BGR2GRAY)
258                    # Detect face in the saved training photo
259                    caras_train = face_cascade_m.detectMultiScale(gris, 1.3, 5)
260                    for (x, y, w, h) in caras_train:
261                        rostro_train = gris[y:y+h, x:x+w]
262                        rostro_train = _cv2.resize(rostro_train, (150, 150))
263                        faces_m.append(rostro_train)
264                        labels_m.append(0)
265                        break # solo la primera cara detectada por foto
266
267        if not faces_m:
268            return {"status": "sin_fotos", "mensaje": "No hay fotos del máster para comparar"}
269
270        recognizer_m.train(faces_m, np.array(labels_m))
271
272        gris_frame = _cv2.cvtColor(frame, _cv2.COLOR_BGR2GRAY)
273        caras = face_cascade_m.detectMultiScale(gris_frame, 1.3, 5)
274
275        for (x, y, w, h) in caras:
276            rostro = gris_frame[y:y+h, x:x+w]
277            rostro = _cv2.resize(rostro, (150, 150))
278            label, conf = recognizer_m.predict(rostro)
279            if conf < 95:
280                return {"status": "verificado", "llave": config.LLAVE_MAESTRA}
281
282        return {"status": "no_reconocido"}
283    except Exception as e:
284        return {"status": "error", "mensaje": str(e)}, 500

662    # 3. Promedio diario de Entradas y Salidas
